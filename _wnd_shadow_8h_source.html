117  // Helper to calculate the alpha-premultiled value for a pixel
118  inline DWORD PreMultiply(COLORREF cl, unsigned char nAlpha)
119  {
120  // It's strange that the byte order of RGB in 32b BMP is reverse to in COLORREF
121  return (GetRValue(cl) * (DWORD)nAlpha / 255) << 16 |
